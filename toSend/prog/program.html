# Cakowanie funkcji f na przedziale [left,right] 
# metod trapezw z points rwnoodlegymi punktami
function int_trapez(f, left, right, points)
    width = (right-left)/(points-1);
    sum = (f(left)+f(right))*0.5;
    
    for i in 1:(points-1)
        sum += f(left + i*width);
    end
    
    sum *= width;
    return sum;
end

function int_trapez_right_to_left(f, left, right, points)
    width = (right-left)/(points-1);
    sum = (f(left)+f(right))*0.5;
    
    for i in (points-1):1
        sum += f(left + i*width);
    end
    
    sum *= width;
    return sum;
end

# Zoona kwadratura Gaussa-Legendre'a na przedziale [0,1]
function wtf(f, div, deg) # div - podzia przedziau [0,1]
    out = 0.0;
    for i in 1:(vlength(div)-1)
        out += gl_base(shift(f,div[i],div[i+1]),deg);
    end
    
    return out
end

function eqd(steps) # podzia odcinka punktami rwnoodlegymi
    return [i*(1/Float64(steps)) for i in 0:steps]
end
